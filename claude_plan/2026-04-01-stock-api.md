```python
# stock-api/cache.py
import time
from typing import Any, Optional

class Cache:
    def __init__(self):
        self._store: dict[str, tuple[Any, float]] = {}

    def get(self, key: str) -> Optional[Any]:
        entry = self._store.get(key)
        if entry is None:
            return None
        value, expires_at = entry
        if time.time() > expires_at:
            del self._store[key]
            return None
        return value

    def set(self, key: str, value: Any, ttl: int) -> None:
        self._store[key] = (value, time.time() + ttl)

    def clear(self, key: str) -> None:
        self._store.pop(key, None)
```

```python
# stock-api/sources.py
import akshare as ak
import pandas as pd
from datetime import datetime, timedelta
from cache import Cache

_cache = Cache()


def is_hk(symbol: str) -> bool:
    """港股：5位数字（可带 .HK 后缀）"""
    s = symbol.upper().replace('.HK', '')
    return len(s) == 5 and s.isdigit()


def normalize_symbol(symbol: str) -> str:
    return symbol.upper().replace('.HK', '')


def _a_market(symbol: str) -> str:
    """A股市场：sh / sz"""
    return 'sh' if symbol.startswith('6') else 'sz'


def get_quote(symbol: str) -> dict:
    s = normalize_symbol(symbol)
    if is_hk(symbol):
        cache_key = 'quote_hk_all'
        fetch = lambda: ak.stock_hk_spot_em()
    else:
        cache_key = 'quote_a_all'
        fetch = lambda: ak.stock_zh_a_spot_em()

    df = _cache.get(cache_key)
    if df is None:
        try:
            df = fetch()
        except Exception as e:
            return {'error': 'data_unavailable', 'message': str(e)}
        _cache.set(cache_key, df, ttl=60)

    row = df[df['代码'] == s]
    if row.empty:
        return {'error': 'invalid_symbol', 'symbol': symbol}

    r = row.iloc[0]
    return {
        'symbol': symbol,
        'name': str(r.get('名称', '')),
        'price': _safe(r, '最新价'),
        'change_pct': _safe(r, '涨跌幅'),
        'volume': _safe(r, '成交额'),
        'pe': _safe(r, '市盈率-动态'),
        'pb': _safe(r, '市净率'),
        'trading': _is_trading_now(),
    }


def get_intraday(symbol: str) -> dict:
    s = normalize_symbol(symbol)
    cache_key = f'intraday:{s}'
    data = _cache.get(cache_key)
    if data is not None:
        return data

    try:
        if is_hk(symbol):
            df = ak.stock_hk_hist_min_em(symbol=s, period='1', adjust='')
        else:
            df = ak.stock_intraday_em(symbol=s)
    except Exception as e:
        return {'error': 'data_unavailable', 'message': str(e)}

    records = df.to_dict(orient='records')
    result = {'symbol': symbol, 'data': records}
    _cache.set(cache_key, result, ttl=30)
    return result


def get_flow(symbol: str) -> dict:
    if is_hk(symbol):
        return {'error': 'unsupported_market', 'message': '港股暂不支持资金流向查询'}

    s = normalize_symbol(symbol)
    cache_key = f'flow:{s}'
    data = _cache.get(cache_key)
    if data is not None:
        return data

    try:
        df = ak.stock_individual_fund_flow(stock=s, market=_a_market(s))
    except Exception as e:
        return {'error': 'data_unavailable', 'message': str(e)}

    if df.empty:
        return {'error': 'data_unavailable', 'message': '无资金流数据'}

    r = df.iloc[-1]  # 最后一行 = 今日累计
    result = {
        'symbol': symbol,
        'date': str(r.get('日期', '')),
        'main_net': _safe(r, '主力净流入-净额'),
        'main_net_pct': _safe(r, '主力净流入-净占比'),
        'super_large_net': _safe(r, '超大单净流入-净额'),
        'large_net': _safe(r, '大单净流入-净额'),
        'retail_net': _safe(r, '小单净流入-净额'),
    }
    _cache.set(cache_key, result, ttl=60)
    return result


def get_news(symbol: str) -> dict:
    s = normalize_symbol(symbol)
    cache_key = f'news:{s}'
    data = _cache.get(cache_key)
    if data is not None:
        return data

    try:
        df = ak.stock_news_em(symbol=s)
    except Exception as e:
        return {'error': 'data_unavailable', 'message': str(e)}

    records = df.head(10).to_dict(orient='records')
    result = {'symbol': symbol, 'news': records}
    _cache.set(cache_key, result, ttl=300)
    return result


def get_announcement(symbol: str) -> dict:
    if is_hk(symbol):
        return {'error': 'unsupported_market', 'message': '港股暂不支持公告查询'}

    s = normalize_symbol(symbol)
    cache_key = f'announcement:{s}'
    data = _cache.get(cache_key)
    if data is not None:
        return data

    end = datetime.today().strftime('%Y%m%d')
    start = (datetime.today() - timedelta(days=30)).strftime('%Y%m%d')
    try:
        df = ak.stock_zh_a_disclosure_report_cninfo(
            symbol=s, market='A股', start_date=start, end_date=end
        )
    except Exception as e:
        return {'error': 'data_unavailable', 'message': str(e)}

    records = df.head(5).to_dict(orient='records')
    result = {'symbol': symbol, 'announcements': records}
    _cache.set(cache_key, result, ttl=300)
    return result


def _safe(row, col: str):
    """安全取值，列不存在时返回 None"""
    try:
        v = row.get(col)
        if pd.isna(v):
            return None
        return v
    except Exception:
        return None


def _is_trading_now() -> bool:
    """判断当前是否为 A股/港股交易时间（北京时间）"""
    from datetime import datetime
    import pytz
    now = datetime.now(pytz.timezone('Asia/Shanghai'))
    if now.weekday() >= 5:  # 周六日
        return False
    t = now.hour * 100 + now.minute
    return (930 <= t <= 1130) or (1300 <= t <= 1500)
```

```python
# stock-api/main.py
from fastapi import FastAPI, Query
import sources

app = FastAPI(title="Stock API", version="1.0")


@app.get('/quote')
def quote(symbol: str = Query(..., description="股票代码，如 600519 或 00700.HK")):
    return sources.get_quote(symbol)


@app.get('/intraday')
def intraday(symbol: str = Query(...)):
    return sources.get_intraday(symbol)


@app.get('/flow')
def flow(symbol: str = Query(...)):
    return sources.get_flow(symbol)


@app.get('/news')
def news(symbol: str = Query(...)):
    return sources.get_news(symbol)


@app.get('/announcement')
def announcement(symbol: str = Query(...)):
    return sources.get_announcement(symbol)
```

```markdown
## 实时行情工具（stock-api）

本地服务运行在 `http://localhost:7070`（需先启动 `stock-api/start.bat` 或已配置开机自启）。

使用 `web_fetch` 调用，无需任何鉴权。

| 用户意图 | 接口 | 示例 |
|---------|------|------|
| 查最新价格/行情 | `GET /quote?symbol={symbol}` | `/quote?symbol=600519` |
| 查分时走势 | `GET /intraday?symbol={symbol}` | `/intraday?symbol=00700.HK` |
| 查资金流向/主力动向 | `GET /flow?symbol={symbol}` | `/flow?symbol=600519`（仅A股）|
| 查相关新闻/消息面 | `GET /news?symbol={symbol}` | `/news?symbol=600519` |
| 查最新公告/披露 | `GET /announcement?symbol={symbol}` | `/announcement?symbol=601398`（仅A股）|

**Symbol 格式：**
```
